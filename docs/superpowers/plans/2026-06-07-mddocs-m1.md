# mddocs M1 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship a local-first, git-native markdown collaboration tool — `mddocs open <file>` gives a browser editor with comments, suggestions, provenance, and git-backed history — built on the MIT `proof-sdk`.

**Architecture:** Fork the `proof-sdk` npm-workspaces monorepo. Keep `@proof/*` packages unchanged and reuse `@proof/core` (marks model, anchoring, `embedMarks`/`extractMarks`) and `@proof/editor` (browser editor). Add two new workspace packages: `mddocs-local` (engine: load/save `.md`↔marks, git, local serve — all TDD) and `mddocs-cli` (thin commander CLI). Collaboration state persists as a `<!-- PROOF -->` JSON footer inside each `.md`; history/sync/multiplayer ride on plain git.

**Tech Stack:** TypeScript, Node 20+, npm workspaces, `tsx`, **vitest** (new — for our packages), `commander` (CLI), `simple-git` (git wrapper), `@proof/core` + `@proof/editor` (reused), Vite (serves the editor).

**Spec:** `docs/superpowers/specs/2026-06-07-mddocs-m1-design.md`

## Conventions used in this plan

- All commands run from the repo root `~/Open-Source Projects/mddocs` unless stated.
- Our two packages live at `packages/mddocs-local` and `packages/mddocs-cli`.
- We never edit files under `packages/doc-*` or root `src/` (the `@proof/*` fork). If we discover we must, stop and escalate.
- Every `@proof/core` function we use is funnelled through one adapter module (`packages/mddocs-local/src/proof.ts`) so the SDK boundary — and any import-path surprises — lives in exactly one file.
- Run a single package's tests with: `npm test -w mddocs-local` (or `-w mddocs-cli`).

```ts
import { readFile, writeFile, rename } from 'node:fs/promises'
import { extractMarks, embedMarks } from './proof'
import type { LoadedDoc } from './types'
import type { StoredMark } from './proof'

export async function loadDoc(path: string): Promise<LoadedDoc> {
  let raw: string
  try {
    raw = await readFile(path, 'utf8')
  } catch (err: unknown) {
    if ((err as NodeJS.ErrnoException).code === 'ENOENT') {
      return { content: '', marks: {} }
    }
    throw err
  }
  const { content, marks } = extractMarks(raw)
  return { content, marks: (marks ?? {}) as Record<string, StoredMark> }
}

export async function saveDoc(
  path: string,
  content: string,
  marks: Record<string, StoredMark>,
): Promise<void> {
  const out = Object.keys(marks).length > 0 ? embedMarks(content, marks) : content
  // atomic write: temp file + rename, unique per call to avoid clobber
  const tmp = `${path}.tmp-${process.pid}-${Math.trunc(performance.now())}`
  await writeFile(tmp, out, 'utf8')
  await rename(tmp, path)
}
```

```ts
import { resolveMark } from './proof'
import type { Mark, StoredMark } from './proof'
import type { ReanchorResult } from './types'

export function reanchorMarks(
  content: string,
  marks: Record<string, StoredMark>,
): ReanchorResult {
  const out: Record<string, StoredMark> = {}
  const orphaned: Mark[] = []
  for (const [id, stored] of Object.entries(marks)) {
    const mark = stored as unknown as Mark
    const resolved = resolveMark(content, mark)
    if (!resolved || resolved.orphaned) {
      const flagged = { ...mark, orphaned: true }
      out[id] = flagged as unknown as StoredMark
      orphaned.push(flagged)
    } else {
      out[id] = { ...mark, range: { from: resolved.from, to: resolved.to }, orphaned: false } as unknown as StoredMark
    }
  }
  return { marks: out, orphaned }
}
```

> Note: `resolveMark` signature confirmed in spec §2.1 as `resolveMark(docText, mark, docSize?) → { from; to; orphaned } | null`. If Task 1's inspection showed a different shape, adjust here and re-run.

```ts
import { extractMarks, embedMarks } from './proof'
import type { StoredMark } from './proof'

const CONFLICT_RE = /^<{7} .*$[\s\S]*?^>{7} .*$/m

export function hasFooterConflict(raw: string): boolean {
  return CONFLICT_RE.test(raw)
}

// Union marks by id; on id collision, keep the one with the later `at`.
function unionMarks(
  a: Record<string, StoredMark>,
  b: Record<string, StoredMark>,
): Record<string, StoredMark> {
  const out: Record<string, StoredMark> = { ...a }
  for (const [id, mark] of Object.entries(b)) {
    const existing = out[id] as unknown as { at?: string } | undefined
    const incoming = mark as unknown as { at?: string }
    if (!existing || (incoming.at ?? '') >= (existing.at ?? '')) out[id] = mark
  }
  return out
}

export function resolveFooterConflictText(raw: string): string {
  const match = raw.match(CONFLICT_RE)
  if (!match) return raw
  const block = match[0]
  // Split the conflict block into the two sides.
  const sep = /^={7}.*$/m
  const [oursRaw, theirsRaw] = block
    .replace(/^<{7} .*$\n?/m, '')
    .replace(/^>{7} .*$\n?/m, '')
    .split(sep)
  const ours = extractMarks(oursRaw ?? '').marks ?? {}
  const theirs = extractMarks(theirsRaw ?? '').marks ?? {}
  const merged = unionMarks(
    ours as Record<string, StoredMark>,
    theirs as Record<string, StoredMark>,
  )
  // Prose = everything before the conflict block, footer rebuilt from union.
  const prose = raw.slice(0, match.index ?? 0).replace(/\s+$/, '')
  return embedMarks(prose, merged)
}
```

```ts
import { createServer, type Server } from 'node:http'
import { loadDoc, saveDoc } from './doc'
import { commitFile, isGitRepo } from './git'
import { reanchorMarks } from './reanchor'
import { basename, dirname } from 'node:path'
import type { StoredMark } from './proof'

export interface SessionOptions { autocommit?: boolean; debounceMs?: number }

export interface Session {
  applyChange(content: string, marks: Record<string, StoredMark>): Promise<void>
  stop(): Promise<void>
}

export async function createSession(path: string, opts: SessionOptions = {}): Promise<Session> {
  const debounceMs = opts.debounceMs ?? 400
  const canCommit = opts.autocommit ? await isGitRepo(dirname(path)) : false
  let commitTimer: NodeJS.Timeout | undefined

  async function applyChange(content: string, marks: Record<string, StoredMark>): Promise<void> {
    const { marks: reanchored } = reanchorMarks(content, marks)
    await saveDoc(path, content, reanchored)
    if (canCommit) {
      if (commitTimer) clearTimeout(commitTimer)
      commitTimer = setTimeout(() => {
        void commitFile(path, `mddocs: edit ${basename(path)}`).catch(() => {})
      }, debounceMs)
    }
  }

  return {
    applyChange,
    async stop() { if (commitTimer) clearTimeout(commitTimer) },
  }
}

// serve(): boots createSession + an HTTP/WS server hosting @proof/editor.
// Editor wiring per packages/mddocs-local/SPIKE-editor.md (Task 2).
export async function serve(
  path: string,
  opts: SessionOptions & { port?: number } = {},
): Promise<{ url: string; stop(): Promise<void>; server: Server }> {
  const session = await createSession(path, opts)
  const initial = await loadDoc(path)
  void initial // handed to the editor bundle at the route built per Task 2
  const server = createServer((_req, res) => {
    // The editor app + a WS endpoint are mounted here following SPIKE-editor.md.
    // WS messages of shape { content, marks } call session.applyChange(...).
    res.statusCode = 200
    res.end('mddocs editor host')
  })
  const port = opts.port ?? 0
  await new Promise<void>((resolve) => server.listen(port, '127.0.0.1', resolve))
  const addr = server.address()
  const actualPort = typeof addr === 'object' && addr ? addr.port : port
  return {
    url: `http://127.0.0.1:${actualPort}`,
    async stop() { await session.stop(); await new Promise<void>((r) => server.close(() => r())) },
    server,
  }
}
```

```ts
import { loadDoc, saveDoc, proof } from 'mddocs-local'
import { fileForId } from '../util/resolve-file'
import type { StoredMark } from 'mddocs-local'

function actor(): string {
  return `human:${process.env.USER ?? 'unknown'}`
}

export function registerComment(program: Command): void {
  const cmd = program.command('comment').description('manage comments')

  cmd.command('add <file>')
    .requiredOption('--quote <q>')
    .requiredOption('--text <t>')
    .action(async (file: string, o: { quote: string; text: string }) => {
      const doc = await loadDoc(file)
      const mark = proof.createComment(o.quote, actor(), o.text, undefined, undefined)
      doc.marks[mark.id] = mark as unknown as StoredMark
      await saveDoc(file, doc.content, doc.marks)
      console.log(`added comment ${mark.id}`)
    })

  cmd.command('ls <file>')
    .option('--open').option('--resolved').option('--orphaned')
    .action(async (file: string, o: { open?: boolean; resolved?: boolean; orphaned?: boolean }) => {
      const doc = await loadDoc(file)
      for (const m of Object.values(doc.marks) as Array<Record<string, unknown>>) {
        if (m.kind !== 'comment') continue
        const data = (m.data ?? {}) as { resolved?: boolean; text?: string }
        if (o.resolved && !data.resolved) continue
        if (o.open && data.resolved) continue
        if (o.orphaned && !m.orphaned) continue
        console.log(`${m.id}  ${data.resolved ? '[resolved]' : '[open]'}${m.orphaned ? ' [orphaned]' : ''}  ${data.text ?? ''}`)
      }
    })

  cmd.command('reply <id>')
    .requiredOption('--text <t>')
    .option('--file <f>')
    .action(async (id: string, o: { text: string; file?: string }) => {
      const file = fileForId(o)
      const doc = await loadDoc(file)
      const updated = proof.getThread(Object.values(doc.marks) as never, id) // validates thread exists
      void updated
      // Replies are stored on the comment's data.replies via @proof/core's model.
      const mark = doc.marks[id] as unknown as { data?: { replies?: unknown[] } }
      mark.data = mark.data ?? {}
      mark.data.replies = [...(mark.data.replies ?? []), { by: actor(), text: o.text }]
      await saveDoc(file, doc.content, doc.marks)
      console.log(`replied to ${id}`)
    })

  cmd.command('resolve <id>')
    .option('--file <f>')
    .action(async (id: string, o: { file?: string }) => {
      const file = fileForId(o)
      const doc = await loadDoc(file)
      const next = proof.resolveComment(Object.values(doc.marks) as never, id)
      doc.marks = Object.fromEntries((next as Array<{ id: string }>).map((m) => [m.id, m as unknown as StoredMark]))
      await saveDoc(file, doc.content, doc.marks)
      console.log(`resolved ${id}`)
    })
}
```

```ts
import { fileForId } from '../util/resolve-file'
import type { StoredMark } from 'mddocs-local'

function apply(
  fn: (marks: never, id: string) => unknown,
  verb: string,
) {
  return async (id: string, o: { file?: string }) => {
    const file = fileForId(o)
    const doc = await loadDoc(file)
    const next = fn(Object.values(doc.marks) as never, id) as Array<{ id: string }>
    doc.marks = Object.fromEntries(next.map((m) => [m.id, m as unknown as StoredMark]))
    await saveDoc(file, doc.content, doc.marks)
    console.log(`${verb} ${id}`)
  }
}

export function registerAcceptReject(program: Command): void {
  program.command('accept <id>').option('--file <f>').action(apply(proof.acceptSuggestion, 'accepted'))
  program.command('reject <id>').option('--file <f>').action(apply(proof.rejectSuggestion, 'rejected'))
}
```

> `acceptSuggestion`/`rejectSuggestion` take `(marks: Mark[], id)` and return `Mark[]` (spec §2.1). Accept rewrites `content`-bearing marks into the doc; in M1 the content rewrite is owned by `@proof/core` on the marks array — if accept must also mutate `doc.content`, the editor path (Task 9) handles live application; the CLI path records the status and the next `open` reconciles. Verify behaviour in Task 1 and tighten this test+code if accept is expected to rewrite prose at the CLI.

```ts
import type { Command } from 'commander'
import { serve, isGitRepo } from 'mddocs-local'
import { dirname } from 'node:path'

export function registerOpen(program: Command): void {
  program.command('open <file>')
    .description('open the markdown file in the browser editor')
    .option('--port <n>', 'port', (v) => parseInt(v, 10))
    .option('--no-autocommit', 'do not auto-commit edits')
    .action(async (file: string, o: { port?: number; autocommit?: boolean }) => {
      if (o.autocommit !== false && !(await isGitRepo(dirname(file)))) {
        console.warn('mddocs: not a git repo — history/autocommit disabled. Run `git init` to enable.')
      }
      const session = await serve(file, { port: o.port, autocommit: o.autocommit !== false })
      console.log(`mddocs: editing ${file} at ${session.url}  (Ctrl-C to stop)`)
      process.on('SIGINT', () => { void session.stop().then(() => process.exit(0)) })
    })
}
```

```ts
    const p = join(dir, 'doc.md')
    const conflicted = [
      '# Shared', '', 'Agreed body.', '',
      '<<<<<<< HEAD',
```
